import { Component } from '@angular/core';
import { Router } from '@angular/router';

/**
 * @description
 * Componente que gestiona el formulario de la cesta de compras, donde el usuario ingresa sus datos para proceder con el pago
 */

@Component({
  selector: 'app-formulario-cesta',
  templateUrl: './formulario-cesta.component.html',
  styleUrls: ['./formulario-cesta.component.css']
})
export class FormularioCestaComponent {
  mensaje: string = ''; // Mensaje de confirmaci√≥n
  formData: any = { // Datos del formulario
    nombre: '',
    email: '',
    telefono: '',
    direccion: '',
    ciudad: '',
    cp: '',
    tarjeta: '',
    expiracion: '',
    cvv: ''
  };

  errores: any = {}; // Errores de validaci√≥n

  /**
   * Constructor del componente.
   * 
   * @param router Router para realizar redirecciones despu√©s de enviar el formulario.
   */
  constructor(private router: Router) {}

  /**
   * M√©todo que valida los campos del formulario.
   * 
   * @returns boolean Retorna true si no hay errores de validaci√≥n, false si hay errores.
   */
  validarFormulario(): boolean {
    this.errores = {};

    // Validar nombre
    if (!this.formData.nombre.trim()) {
      this.errores.nombre = 'El nombre es obligatorio';
    }

    // Validar email
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!this.formData.email.trim()) {
      this.errores.email = 'El correo electr√≥nico es obligatorio';
    } else if (!emailRegex.test(this.formData.email)) {
      this.errores.email = 'Correo electr√≥nico no v√°lido';
    }

    // Validar tel√©fono
    const telefonoRegex = /^[0-9]{9,12}$/;
    if (!this.formData.telefono.trim()) {
      this.errores.telefono = 'El tel√©fono es obligatorio';
    } else if (!telefonoRegex.test(this.formData.telefono)) {
      this.errores.telefono = 'N√∫mero de tel√©fono inv√°lido';
    }

    // Validar direcci√≥n
    if (!this.formData.direccion.trim()) {
      this.errores.direccion = 'La direcci√≥n es obligatoria';
    }

    // Validar ciudad
    if (!this.formData.ciudad.trim()) {
      this.errores.ciudad = 'La ciudad es obligatoria';
    }

    // Validar c√≥digo postal
    const cpRegex = /^[0-9]{4,6}$/;
    if (!this.formData.cp.trim()) {
      this.errores.cp = 'El c√≥digo postal es obligatorio';
    } else if (!cpRegex.test(this.formData.cp)) {
      this.errores.cp = 'C√≥digo postal inv√°lido';
    }

    // Validar n√∫mero de tarjeta
    const tarjetaRegex = /^[0-9]{16}$/;
    if (!this.formData.tarjeta.trim()) {
      this.errores.tarjeta = 'El n√∫mero de tarjeta es obligatorio';
    } else if (!tarjetaRegex.test(this.formData.tarjeta)) {
      this.errores.tarjeta = 'N√∫mero de tarjeta inv√°lido';
    }

    // Validar fecha de expiraci√≥n
    if (!this.formData.expiracion.trim()) {
      this.errores.expiracion = 'La fecha de expiraci√≥n es obligatoria';
    }

    // Validar CVV
    const cvvRegex = /^[0-9]{3,4}$/;
    if (!this.formData.cvv.trim()) {
      this.errores.cvv = 'El CVV es obligatorio';
    } else if (!cvvRegex.test(this.formData.cvv)) {
      this.errores.cvv = 'CVV inv√°lido';
    }

    return Object.keys(this.errores).length === 0;
  }

  /**
   * M√©todo que se ejecuta al enviar el formulario.
   * Si el formulario es v√°lido, muestra un mensaje de √©xito y redirige al usuario a la p√°gina de inicio.
   */
  onSubmit() {
    if (this.validarFormulario()) {
      this.mensaje = 'Pedido realizado correctamente';

      // Redirigir despu√©s de unos segundos
      setTimeout(() => {
        this.router.navigate(['/inicio']);
      }, 2000);
    }
  }
}
